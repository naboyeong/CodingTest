from collections import defaultdict, deque
n, m, v = map(int, input().split())
dic = defaultdict(list)
for _ in range(m):
    a, b = map(int, input().split())
    dic[a].append(b)
    dic[b].append(a)

for i in dic.keys():
    dic[i].sort(reverse=True)

stack = deque()
check = set()
stack.append(v)

while(stack):

    current = stack.pop()
    if current in check:
        continue
    check.add(current)

    print(current, end=" ")

    for i in dic[current]:
        if i not in check:
            stack.append(i)
